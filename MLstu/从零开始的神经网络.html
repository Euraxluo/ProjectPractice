## layer
def init_parameter_b(layer):
    assert layer<len(parameter_placeholder),"layer begin from 0,smaller than layers length"
    dist = parameter_placeholder[layer]['b']#b,b0:684,b1:10
    return np.random.rand(input_output_dimensions[layer])*(dist[1]-dist[0])+dist[0]#0..1'b'.value

def init_parameter_w(layer):
    assert layer>0 and layer<len(parameter_placeholder),"layer begin from 1,and smaller than layers length,because w param is not in the layer 0"
    dist = parameter_placeholder[layer]['w']#w,w:684,10
    return np.random.rand(input_output_dimensions[layer-1],input_output_dimensions[layer])*(dist[1]-dist[0])+dist[0]#0..1'b'.value

def init_all_paramters()->list:
    layers = []
    for layer in range(len(parameter_placeholder)):
        layer_parameter={}
        for paramter in parameter_placeholder[layer].keys():
            if paramter=='b':
                layer_parameter['b']=init_parameter_b(layer)
                continue
            if paramter=='w':
                layer_parameter['w']=init_parameter_w(layer)
                continue
        layers.append(layer_parameter)
    return layers

def simple_network(img,paramters):#
    l_0_in = img+paramters[0]['b']#
    l_0_out = Each_layer_activation_function[0](l_0_in)
    l_1_in = np.dot(l_0_out,paramters[1]['w'])+paramters[1]['b']#
    l_1_out = Each_layer_activation_function[1](l_1_in)#output
    return l_1_out
